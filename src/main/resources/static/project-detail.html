        // Fetch project analytics
        async function fetchProjectAnalytics() {
            try {
                const response = await fetch(`/api/v1/projects/${projectId}/analytics`, {
                    headers: {
                        'Authorization': `Bearer ${token}`
                    }
                });

                if (!response.ok) {
                    throw new Error('Failed to fetch project analytics');
                }

                const analytics = await response.json();
                
                // Update stats
                document.getElementById('totalEvents').textContent = analytics.totalEvents;
                document.getElementById('uniqueUsers').textContent = analytics.uniqueUsers;
                document.getElementById('avgSession').textContent = `${analytics.avgSessionDuration}m`;
                document.getElementById('conversionRate').textContent = `${analytics.conversionRate}%`;

                // Update charts
                userActivityChart.data.labels = analytics.userActivity.labels;
                userActivityChart.data.datasets[0].data = analytics.userActivity.data;
                userActivityChart.update();

                eventDistributionChart.data.datasets[0].data = [
                    analytics.eventDistribution.clicks,
                    analytics.eventDistribution.moves,
                    analytics.eventDistribution.scrolls
                ];
                eventDistributionChart.update();
            } catch (error) {
                console.error('Error fetching project analytics:', error);
            }
        }

        // Handle edit button
        document.getElementById('editButton').addEventListener('click', (e) => {
            e.preventDefault();
            window.location.href = `/projects/${projectId}/edit`;
        });

        // Handle delete button
        document.getElementById('deleteButton').addEventListener('click', async () => {
            if (!confirm('Are you sure you want to delete this project? This action cannot be undone.')) {
                return;
            }

            try {
                const response = await fetch(`/api/v1/projects/${projectId}`, {
                    method: 'DELETE',
                    headers: {
                        'Authorization': `Bearer ${token}`
                    }
                });

                if (response.ok) {
                    window.location.href = '/projects';
                } else {
                    const error = await response.json();
                    alert(error.message || 'Failed to delete project');
                }
            } catch (error) {
                alert('An error occurred. Please try again.');
            }
        });

        // Handle copy code button
        document.getElementById('copyCodeButton').addEventListener('click', () => {
            const code = document.getElementById('integrationCode').textContent;
            navigator.clipboard.writeText(code).then(() => {
                alert('Integration code copied to clipboard!');
            }).catch(() => {
                alert('Failed to copy code. Please try again.');
            });
